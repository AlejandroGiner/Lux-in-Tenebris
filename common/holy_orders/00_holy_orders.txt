jesuit_order = {
    icon = GFX_holy_order_jesuits
    trigger = {
		OR = {
			technology_group = western
			capital_scope = { region = carpathia_region }
			capital_scope = { region = poland_region }
			colonial_parent = {
				OR = {
					technology_group = western
					capital_scope = { region = carpathia_region }
					capital_scope = { region = poland_region }
				}
			}
		}
		religion = catholic
	}
    cost = 60
    cost_type = adm_power
    per_province_effect = {
        add_base_tax = 1
    }
    per_province_abandon_effect = {
        add_base_tax = -1
    }
    modifier = {
        local_missionary_strength = 0.015
        local_build_cost = -0.1
    }
    ai_priority = {
        factor = 2
        modifier = {
            factor = 2.0
            NOT = { religion = ROOT }
        }
    }
}

					capital_scope = { region = poland_region }
				}
			}
		}
		religion = catholic
	}
    cost = 60
    cost_type = dip_power
    per_province_effect = {
        add_base_production = 1

        tooltip = {
            custom_tooltip = removes_slaves_tooltip
        }
        hidden_effect = {
            if = {
                limit = {
                    trade_goods = slaves
                }
                change_trade_goods = random
            }
        }
    }
    per_province_abandon_effect = {
        add_base_production = -1
    }
    modifier = {
        local_culture_conversion_cost = -0.3
    }
    ai_priority = {
        factor = 1
        modifier = {
            factor = 2.0
            trade_goods = slaves
        }
        modifier = {
            factor = 2.0
            NOT = { culture = ROOT }
        }
    }
}
