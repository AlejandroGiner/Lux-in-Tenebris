pom_recon_wlg = {
	slot = 1
	#generic = yes
	ai = yes
	
	potential = {
		OR = {
			tag = STP #Stolp
			tag = STE #Stettin
		}
	}
	has_country_shield = yes
		
	pom_gen_recon_wlg = {
		icon = mission_invade_island
		required_missions = { }
		position = 1
		
		provinces_to_highlight = {
			owned_by = WOL
			NOT = { owned_by = ROOT }
		}
		
		trigger = {
			NOT = { exists = WOL }
			owns = 2994 #Wolgast
		}
		
		effect = {
			add_mil_power = 50
			add_adm_power = 50
			add_prestige_or_monarch_power = { amount = 10 }
			#Claims
			if = {
				limit = {
					NOT = { tag = STE }
				}
				custom_tooltip = pom_gen_recon_stt_claims
				hidden_effect = {
					every_province = {
						limit = {
							owned_by = STE
							NOT = { is_core = ROOT }
							NOT = { owned_by = ROOT }
						}
						add_permanent_claim = ROOT
					}
				}
			}
			if = {
				limit = {
					NOT = { tag = STP }
				}
				custom_tooltip = pom_gen_recon_stp_claims
				hidden_effect = {
					every_province = {

						}
