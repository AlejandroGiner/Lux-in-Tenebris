cambodian_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = KHM
			tag = KME
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	khm_reclaim_old_cambodia = {
		icon = mission_unite_home_region
		required_missions = {  }
		provinces_to_highlight = {
			OR = {
				province_id = 4830
				province_id = 5957
				province_id = 603
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			4830 = { owned_by = ROOT }
			5957 = { owned_by = ROOT }
			603 = { owned_by = ROOT }
		}
		effect = {
			4830 = { change_culture = ROOT }
			5957 = { change_culture = ROOT }
			603 = { change_culture = ROOT }
			central_thai_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			gulf_siam_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	khm_conquer_the_siamese_heartland = {
		icon = mission_build_up_to_force_limit
		required_missions = { khm_reclaim_old_cambodia }
		position = 5
		provinces_to_highlight = {
			OR = {
				area = central_thai_area
				area = gulf_siam_area
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			central_thai_area = {
				type = all
				owned_by = ROOT
			}
			gulf_siam_area = {
				type = all
				owned_by = ROOT
			}
		}
		effect = {
			sukhothai_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	khm_end_siam = {
		icon = mission_conquer_siam
		required_missions = { khm_conquer_the_siamese_heartland }
		provinces_to_highlight = {
			area = sukhothai_area
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			sukhothai_area = {
				type = all
				owned_by = ROOT
			}
		}
		effect = {
			add_prestige = 50
			add_adm_power = 100
		}
	}
}
