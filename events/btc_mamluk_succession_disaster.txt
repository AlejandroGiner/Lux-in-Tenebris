			}
		}
		random_owned_province = {
			limit = {

			}
			spawn_rebels = {
				type = mamluk_rebels
				size = 1
			}
			set_province_flag = mamluk_rebels_here
		}
		random_owned_province = {
			limit = {
				NOT = { is_capital = yes }
				area = bahari_area #Lower Egypt
				NOT = { has_province_flag = mamluk_rebels_here }
			}
			spawn_rebels = {
				type = mamluk_rebels
				size = 1
			}
		}
		bahari_area = {
			clr_province_flag = mamluk_rebels_here
		}
	}
}

#Mamluks Mutiny in [province]
country_event = {
	id = btc_mamluk_succession_disaster.4
	title = btc_mamluk_succession_disaster.4.t
	desc = btc_mamluk_succession_disaster.4.d
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		any_owned_province = {
			is_capital = no
			OR = {
				region = egypt_region
				region = mashriq_region
				#Iraq
				area = basra_area
				area = mesopotamian_marshes_area
				area = iraq_arabi_area
				area = al_jazira_area
				area = dulaim_area
				area = kut_area
				area = diwaniya_area
				area = rakka_area
				area = palmyra_area
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					OR = {
						region = egypt_region
						region = mashriq_region
						#Iraq
						area = basra_area
						area = mesopotamian_marshes_area
						area = iraq_arabi_area
						area = dulaim_area
						area = kut_area
						area = diwaniya_area
						area = al_jazira_area
						area = rakka_area
						area = palmyra_area
					}
				}
				save_event_target_as = mamluk_mutiny_province
			}
		}
	}
	
	option = {
		name = btc_mamluk_succession_disaster.4.a
		event_target:mamluk_mutiny_province = {
			spawn_rebels = {
				type = mamluk_rebels
				size = 1
			}
		}
	}
	
}
