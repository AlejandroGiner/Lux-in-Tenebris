#Potato spread events, coded by Alejandro, written by Dynamo and designed by Helena
#Potato arrival event
country_event = {
  id = btc_tradegoods.1
  title = btc_tradegoods.2.t
  desc = btc_tradegoods.2.d
  picture = CONQUISTADORS_eventPicture
  
  fire_only_once = yes
  
  trigger = {
    capital_scope = {
      OR = {
        region = france_region
        region = low_countries_region
        region = north_german_region
        region = south_german_region
        region = iberia_region
        region = british_isles_region
        region = ireland_region
      }
    }
    has_global_flag = columbian_exchange_happened
    this_or_colony_has_goods = {
		GOODS = potatoes
	}
	if = {
		limit = {
			3304 = { owned_by = CAS }
			macaronesia_area = { owned_by = CAS }
			CAS = {
				this_or_colony_has_goods = {
					GOODS = potatoes
				}
			}
		}
		tag = CAS
	}
	else_if = {
		limit = {
			3304 = { owned_by = SPA }
			macaronesia_area = { owned_by = SPA }
			SPA = {
				this_or_colony_has_goods = {
					GOODS = potatoes
				}
			}
		}
		tag = SPA
	}
  }
  
  mean_time_to_happen = {
    years = 20
  }
  
  immediate = {
  	hidden_effect = {
		set_global_flag = potatoes_arrival
	}
  }
  
  option = {
    name = btc_tradegoods.2.customloc
	if = {
		limit = {
			OR = { tag = CAS tag = SPA }
			macaronesia_area = { owned_by = ROOT }
		}
		random_owned_province = {
			limit = {
				area = macaronesia_area
			}
			add_permanent_province_modifier = {
				name = potato_port
				duration = -1
	        }
			change_trade_goods = potatoes
			add_base_production = 2
			add_base_manpower = 1
		}
	}
	else = {
		random_owned_province = {
			limit = {
				has_port = yes
				continent = europe
				province_has_center_of_trade_of_level = 1
				province_without_special_goods_produced_trigger = yes
			}
			add_permanent_province_modifier = {
				name = potato_port
				duration = -1
			}
			change_trade_goods = potatoes
			add_base_production = 2
			add_base_manpower = 1
		}
	}
	random_owned_province = {
		limit = {
			province_without_special_goods_produced_trigger = yes
			province_eligible_for_potato_spawn = yes
		}
		change_trade_goods = potatoes
		add_base_production = 1
		add_base_manpower = 1
	}
	change_price = {
		trade_goods = potatoes
		key = btc_tradegoods.2.t
		value = 0.333 #2.25 -> 3.0
		duration = -1
	}
  }
}

      }
    }
    has_global_flag = columbian_exchange_happened
    this_or_colony_has_goods = {
		GOODS = maize

		limit = {

				capital_scope = { region = iberia_region }
				this_or_colony_has_goods = {
					GOODS = maize
				}
			}
		}

  }
  
  mean_time_to_happen = {
    years = 20

    hidden_effect = {

  option = {
    name = btc_tradegoods.6.a
	random_owned_province = {
      limit = {
        has_port = yes
        continent = europe
        province_has_center_of_trade_of_level = 1
        province_without_special_goods_produced_trigger = yes
      }
      add_permanent_province_modifier = {
        name = maize_port
        duration = -1
      }
	  change_trade_goods = maize
    }
	random_owned_province = {
		limit = {
			province_without_special_goods_produced_trigger = yes
			province_eligible_for_potato_spawn = yes
		}
		change_trade_goods = maize
        add_base_production = 1
        add_base_manpower = 1

	change_price = {
      trade_goods = maize

#Macael Marble, coded, written and designed by Helena
province_event = {
	id = btc_tradegoods.9
	title = btc_tradegoods.9.t
	desc = btc_tradegoods.9.d
	picture = CITY_DEVELOPMENT_eventPicture

	fire_only_once = yes

	trigger = {
		province_id = 3145
		is_state = yes
		renaissance = 100
		owner = { 
			has_institution = renaissance
			OR = {
				tag = CAS
				tag = ARA
				tag = SPA
				tag = GRA
			}
		}
		NOT = { devastation = 1 }
		has_global_flag = massa_marble
	}

	mean_time_to_happen = {
		years = 20
	}
  
	option = {
		name = btc_tradegoods.9.a
		owner = { add_prestige_or_monarch_power = { amount = 5 } }
		add_base_production = 1
		change_trade_goods = marble
	}
}
