namespace = BtPLegacyUpperNileEvents

#Pressure from the Shilluk
country_event = {
	id = BtPLegacyUpperNileEvents.1
	title = "BtPLegacyUpperNileEvents.EVTNAME1"
	desc = "BtPLegacyUpperNileEvents.EVTDESC1"
	picture = NATIVES_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FNJ
		5290 = { owned_by = FNJ }
		2800 = { is_empty = yes }
		is_year = 1500
	}

	mean_time_to_happen = {
		months = 24
	}
	
	option = { # Allow ourselves to be pushed north
		name = "BtPLegacyUpperNileEvents.EVTOPTA1"
		ai_chance = { factor = 95 }
		2800 = {
			cede_province = FNJ
			add_core = FNJ
			change_culture = luo_culture
			change_religion = shamanism
			add_base_tax = 1
			add_base_production = 1
		}
		set_capital = 2800
		5290 = {
			cede_province = SLK
			add_core = SLK
			remove_core = FNJ
			rename_capital = "Fashoda"
		}
		hidden_effect = {
			SLK = {
				define_ruler = {
					name = "Nyikaango I"
					dynasty = "Shilluk" 
					age = 25
					adm = 5
					dip = 5
					mil = 5
					fixed = yes
				}
			}
		}
		add_prestige = -25
		add_country_modifier = {
			name = "btpl_funj_momentum"
			duration = 7300
		}
	}

	option = { # Resist the Shilluk
		name = "BtPLegacyUpperNileEvents.EVTOPTB1"
		ai_chance = { factor = 5 }
		add_prestige_or_monarch_power = { amount = 10 }
		5290 = { add_core = SLK }
		capital_scope = {
			spawn_rebels = {
				type = nationalist_rebels
				leader = "Nyikaango"
				size = 1
			}
		}
	}
}

#Arrival of Abdullah
country_event = {
	id = BtPLegacyUpperNileEvents.2
	title = "BtPLegacyUpperNileEvents.EVTNAME2"
	desc = "BtPLegacyUpperNileEvents.EVTDESC2"
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = ALO
		2798 = { owned_by = ALO }
		num_of_cities = 2
		NOT = { religion_group = muslim }
		is_year = 1475
	}

	mean_time_to_happen = {
		months = 32
	}
	
	option = { # Uh oh
		name = "BtPLegacyUpperNileEvents.EVTOPTA2"
		ai_chance = { factor = 95 }
		every_owned_province = {
			add_claim = ABD
		}
		set_capital = 5285
		2798 = {
			cede_province = ABD
			add_core = ABD
		}
		hidden_effect = {
			ABD = {
				define_ruler = {
					name = "Abdullah Jamma"
					dynasty = "Abdallab" 
					age = 25
					male = yes
					adm = 5
					dip = 5
					mil = 2
					fixed = yes
				}
				change_religion = sunni
				add_piety = 0.30
				add_treasury = 300
				add_country_modifier = {
					name = "btpl_abdullahs_coffers"
					duration = 1825
				}
				add_country_modifier = {
					name = "btpl_spread_of_islam_in_alodia"
					duration = 7300
				}
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
					}
				}
			}
			egypt_region = {discover_country = ABD}
			horn_of_africa_region = {discover_country = ABD}
			
		}
		ABD = {
			declare_war_with_cb = {
				who = ALO
				casus_belli = cb_conquest
				war_goal_province = 5285
			}
		}
	}

	option = { # Accept it as inevitable
		name = "BtPLegacyUpperNileEvents.EVTOPTB2"
		ai_chance = { factor = 5 }
		define_ruler = {
			name = "Abdullah Jamma"
			dynasty = "Abdallab" 
			age = 25
			adm = 5
			dip = 5
			mil = 2
			fixed = yes
		}
		change_religion = sunni
		add_piety = 0.30
		add_country_modifier = {
			name = "btpl_spread_of_islam_in_alodia"
			duration = 3650
		}
		add_stability = -1
		add_prestige = -50
		add_treasury = 300
	}
}

#Funj Expand Northwards
country_event = {
	id = BtPLegacyUpperNileEvents.3
	title = "BtPLegacyUpperNileEvents.EVTNAME3"
	desc = "BtPLegacyUpperNileEvents.EVTDESC3"
	picture = NATIVES_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FNJ
		has_country_modifier = btpl_funj_momentum
		OR = {
			2799 = { owned_by = ALO }
			2799 = { owned_by = ABD }
		}
	}

	mean_time_to_happen = {
		months = 36
	}
	
	option = { # Make use of this momentum
		name = "BtPLegacyUpperNileEvents.EVTOPTA3"
		ai_chance = { factor = 95 }
		capital_scope = { build_to_forcelimit = { infantry = 1.0 } }
		add_country_modifier = {
			name = "btpl_funj_militarisation"
			duration = 7300
		}
		if = {
			limit = { 2799 = { owned_by = ALO } }
			ALO = {every_owned_province = {add_claim = FNJ}}
			declare_war_with_cb = {
				who = ALO
				casus_belli = cb_conquest
				war_goal_province = 5285
			}
		}
		if = {
			limit = { 2799 = { owned_by = ALO }}
			ABD = {every_owned_province = {add_claim = FNJ}}
			declare_war_with_cb = {
				who = ABD
				casus_belli = cb_conquest
				war_goal_province = 5285
			}
		}
	}
	
	option = { # We dont need to expand to be great
		name = "BtPLegacyUpperNileEvents.EVTOPTB3"
		ai_chance = { factor = 5 }
		remove_country_modifier = btpl_funj_momentum
		add_country_modifier = {
			name = "btpl_funj_demilitarisation"
			duration = 3650
		}
	}
}
