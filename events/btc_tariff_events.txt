# Tariffs on Beer
country_event = {
	id = btc_colonial_tariffs.182
	title = "btc_colonial_tariffs.EVTNAME182"
	desc = "btc_colonial_tariffs.EVTDESC182"
	picture = TRADE_GOODS_PLANTATION_GOODS_IN_WAREHOUSE_eventPicture

	is_triggered_only = yes
	
	trigger = {
		OR = {
			NOT = { has_country_flag = tariffs_on_beer }
			had_country_flag = { flag = tariffs_on_beer days = 7300 }
		}
		any_subject_country = {
			is_colonial_nation_of = ROOT
			any_owned_province = {
				trade_goods = beer
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				beer = 3
			}
		}
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				beer = 5
			}
		}
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				beer = 7
			}
		}
	}
	
	option = {
		name = "colonial_nation.EVTOPTA100"	# Tax it!
		random_country = {
			limit = {
				is_colonial_nation_of = ROOT
				any_owned_province = {
					trade_goods = beer
				}
			}
			tooltip = {
				add_tariff_value = 0.05
			}
			hidden_effect = {
				country_event = { id = btc_colonial_tariffs.183 days = 1 }
			}
		}
		set_country_flag = tariffs_on_beer
	}
	option = {
		name = "btc_colonial_tariffs.EVTOPTB100"	# Better leave it be
		set_country_flag = tariffs_on_beer
		random_list = {
			33 = { add_adm_power = 10 }
			33 = { add_dip_power = 10 }
			33 = { add_mil_power = 10 }
		}
	}
}

# Tariffs on Livestock
country_event = {
	id = btc_colonial_tariffs.218
	title = "btc_colonial_tariffs.EVTNAME218"
	desc = "btc_colonial_tariffs.EVTDESC218"
	picture = TRADE_GOODS_PLANTATION_GOODS_IN_WAREHOUSE_eventPicture

	is_triggered_only = yes
	
	trigger = {
		OR = {
			NOT = { has_country_flag = tariffs_on_livestock }
			had_country_flag = { flag = tariffs_on_livestock days = 7300 }
		}
		any_subject_country = {
			is_colonial_nation_of = ROOT
			any_owned_province = {
				trade_goods = livestock
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				livestock = 3
			}
		}
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				livestock = 5
			}
		}
		modifier = {
			factor = 1.25
			any_subject_country = {
				is_colonial_nation_of = ROOT
				livestock = 7
			}
		}
	}
	
	option = {
		name = "colonial_nation.EVTOPTA100"	# Tax it!
		random_country = {
			limit = {
				is_colonial_nation_of = ROOT
				any_owned_province = {
					trade_goods = livestock
				}
			}
			tooltip = {
				add_tariff_value = 0.05
			}
			hidden_effect = {
				country_event = { id = btc_colonial_tariffs.219 days = 1 }
			}
		}
		set_country_flag = tariffs_on_livestock
	}
	option = {
		name = "btc_colonial_tariffs.EVTOPTB100"	# Better leave it be
		set_country_flag = tariffs_on_livestock
		random_list = {
			33 = { add_adm_power = 10 }
			33 = { add_dip_power = 10 }
			33 = { add_mil_power = 10 }
		}
	}
}
